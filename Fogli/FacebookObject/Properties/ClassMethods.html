def property(*names)
  options = names.last
  options = {} if !options.kind_of?(Hash)

  names.each do |name|
    next if name.kind_of?(Hash)
    name = name.to_s.downcase.to_sym
    properties[name] = options

    # Create a method for reading the property.
    define_method(name) { read_property(name) }
  end
end
